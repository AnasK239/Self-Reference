### string methods

.length()  (counts spaces)

.charAt(x)  ( char at x )

.indexOf("x")   ( index of x )

.lastIndexOf("x")  ( last index of x )

.toUpperCase();    ( force upper case )

.toLowerCase();    ( force lower case )

.trim();          ( trims white space before and after the string )

.replace("x","y");     ( replace all x with y )

.isEmpty();         (boolean)

.contains("x")      ( boolean to check for a charachter )

.equals("x")   (case senstive)   (if the given string equals x (boolean) )

.equalsIgnoreCase (case insenstive)

.substring(start,end)    (end is exclusive) (without ending index the its calculated from start till the end)
